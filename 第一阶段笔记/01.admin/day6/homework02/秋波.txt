案例1:指定yum软件源
为 server0 指定可用的 yum 软件源
C YUM软件库的地址为 http://classroom.example.com/content/rhel7.0/x86_64/dvd/
C 将此配置为虚拟机 server0 的默认软件仓库
C 确认可用的仓库列表
C 利用yum仓库安装xeyes
vim /etc/yum.repos.d/dey.repo
[no]
name=nononon
enabled=1
gpgcheck=0
baseurl=http://classroom.example.com/content/rhel7.0/x86_64/dvd/
~           yum search
          yum install xeyes  M

案例3：chown归属设置
    1）新建/tarena1目录
  mkdir /tarenal 
	a）将属主设为gelin01，属组设为tarena组
useradd gai [tar]
chown :tar /tarenal
gpasswd -a gei tar
	b）使用户gelin01对此目录具有rwx权限，其他人对此目录无任何权限
setfacl -m u:gai:rwx /tarenal
[root@server0 ~]# getfacl /tarenal
getfacl: Removing leading '/' from absolute path names
# file: tarenal
# owner: root
# group: tar
user::rwx
user:gai:rwx
group::r-x
mask::rwx
other::r-x

    2）使用户gelin02能进入、查看/tarena1文件夹（提示：将gelin02加入所属组）
 useradd g02 
 gpasswd -a g02 tar
    3）新建/tarena2目录
	a）将属组设为tarena
	b）使tarena组的任何用户都能在此目录下创建、删除文件
mkdir /tarena2
chown :tar /tarena2
[root@server0 ~]# ls -ld /tarena2
drwxr-xr-x. 2 root tar 6 7月   5 19:17 /tarena2

chmod g=wrx /tarena2

   
    4）新建/tarena/public目录
	a）使任何用户对此目录都有rwx权限
	b）拒绝zhangsan进入此目录，对此目录无任何权限


案例4:权限设置
	1、创建文件夹/data/test,设置目录的访问权限，使所有者和所属组具备读写执行的权限；其他人无任何权限。
 mkdir -p /da/te
chmod gu=wrx,o=--- /da/te 
ls -ld /da/te
	2、递归修改文件夹/data/test的归属使所有者为zhangsan，所属组为tarena。
   chown -R zhangsan:tar /da
	3、请实现在test目录下，新建的所有子文件或目录的所属组都会是tarena。
   chmod g+s /da/te
  
	4、为lisi创建ACL访问权限，使得lisi可以查看/etc/shadow文件
    useradd lisi 
  setfacl -m u:lisi:rx /etc/shadow
