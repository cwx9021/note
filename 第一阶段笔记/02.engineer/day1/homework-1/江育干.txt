案例2：复制、粘贴、移动
 以root用户新建/example/目录，在此目录下新建nsd.txt文件，并进一步完成下列操作
# mkdir /example
# touch /example/nsd.txt

 1）将“I love Study”写入到文件nsd.txt 
# echo I love Study >> /example/nsd.txt
# cat /example/nsd.txt

 2）将nsd.txt重命名为mylove.txt
# mv /example/nsd.txt /example/mylove.txt
# cat /example/mylove.txt

 3）将/etc/passwd、/boot、/etc/group同时拷贝到/example/目录下
# cp -r /etc/passwd /boot /etc/group /example
# ls /example

 4）将ifconfig命令的前两行内容，追加写入mylove.txt
# ifconfig | head -2 >> /example/mylove.txt
# cat //example/mylove.txt

 5）将主机名永久配置文件，拷贝到/example/目录下
# cp  /root/.bashrc /example
# ls /example

 6）将DNS永久配置文件，拷贝到/example/目录下
# cp /etc/resolv.conf /example
# ls /example

 7）将开机自动挂载配置文件，拷贝到/example/目录下
# cp /etc/fstab /example
# ls /example

案例5:虚拟机 server0操作，构建 LVM 存储

C 新建一个名为 systemvg 的卷组 
# fdisk /dev/vdc
# partprobe 
# lsblk
# vgcreate systemvg /dev/vdc[6-7]

C 在此卷组中创建一个名为 vo 的逻辑卷，大小为8G 
# lvcreate -n vo -L 8G systemvg
# lvs

C 将逻辑卷 vo 格式化为 xfs 文件系统 
# mkfs.xfs /dev/systemvg/vo
# blkid /dev/systemvg/vo

C 将逻辑卷 vo 挂载到 /vo 目录，并在此目录下建立一个测试文件 votest.txt，内容为“I AM KING.” 
# mkdir /vo
# vim /etc/fstab
# mount -a
# df -h
# touch /vo/votest.txt
# echo I AM KING > /vo/votest.txt
# cat /vo/votest.txt

C 实现逻辑卷vo开机自动挂载到/vo
# umount /vo
# mount /dev/systemvg/vo /vo
# mount -a
# df -h

案例8: vim文本编辑器练习，命令模式下的基本操作（重点知识）

 1）将/etc/passwd的前20行，写入到文件/opt/pass20.txt 
# vim /etc/passwd
 20+yy
# touch /opt/pass20.txt
# vim /opt/pass20.txt
 p
# cat /opt/pass20.txt

 2）将/etc/shadow的前10行,追加写入到文件/opt/pass20.txt 
# head -10 /etc/shadow |>> /opt/pass20.txt
# cat /opt/pass20.txt


 3）利用/opt/pass20.txt，进项如下操作：
    C 命令模式下移动光标：键盘上下左右键、Home键、End键
    C 命令模式下行间跳转：到全文的第一行（1G或gg）、到全文的最后一行（G）、到全文的第10行（10G）
    
    C 命令模式下复制、粘贴：
    	 复制1行（yy）、复制3行（3yy）
    	 粘贴到当前行之后（小写p）

    C 命令模式下删除：
    	 删除单个字符（x）
    	 删除到行首（d^）、删除到行尾（d$）
    	 删除1行（dd）、删除3行（3dd）
         
    
    C 命令模式下查找关键词： 
    	 搜索（/word）切换结果（n、N）

    C 补充：在命令模式下大写的C，可以删除光标之后，并且进入输入模式
