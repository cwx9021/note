案例1:为虚拟机 server 配置以下虚拟Web主机
- 实现三个网站的部署
- 实现客户端访问server0.example.com网页内容为  复仇者联盟4
- 实现客户端访问www0.example.com网页内容为 琅琊榜30
- 实现客户端访问webapp0.example.com网页内容为 38里春风不如你

=[root@server0 ~]# vim /etc/httpd/conf.d/1234.conf
<VirtualHost *:80>
  ServerName  server0.example.com
  DocumentRoot /var/www/nsd1
</VirtualHost>
<VirtualHost *:80>
  ServerName  www0.example.com
  DocumentRoot /var/www/nsd2
</VirtualHost>
<VirtualHost *:80>
  ServerName  webapp0.example.com
  DocumentRoot /var/www/nsd3
</VirtualHost>
[root@server0 ~]# mkdir /var/www/nsd1
[root@server0 ~]# mkdir /var/www/nsd2
[root@server0 ~]# mkdir /var/www/nsd3
[root@server0 ~]# ls /var/www/
cgi-bin  html  myweb  nsd1  nsd2  nsd3
[root@server0 ~]# echo '<h1>复仇者联盟4' > /var/www/nsd1/index.html
[root@server0 ~]# echo '<h1>琅琊榜30' > /var/www/nsd2/index.html
[root@server0 ~]# echo '<h1>38里春风不如你' > /var/www/nsd3/index.html
[root@server0 ~]# systemctl restart httpd



案例2:为虚拟机 server 配置Web访问控制
在 Web 网站 http://server0.example.com 的 DocumentRoot 目录下创建一个名为 private 的子目录，要求如下：
1）在server0.example.com的DocumentRoot目录下，private的子目录里书写网页文件index.html内容为  不许偷看丫
2）此页面只能在本机浏览，但是从其他系统不能访问这个目录的内容

=[root@server0 ~]# vim /etc/httpd/conf.d/nsd01.conf
   <VirtualHost *:80>
   ServerName server0.example.com
   DocumentRoot /var/www/myweb
 </VirtualHost>

[root@server0 ~]# mkdir /var/www/myweb/private
[root@server0 ~]# echo '<h1>不许偷看Y' > /var/www/myweb/private/index.html
[root@server0 ~]# vim /etc/httpd/conf.d/nsd02.conf
<Directory  /var/www/myweb/private>
    Require ip 172.25.0.11      
 </Directory>
[root@server0 ~]# systemctl restart httpd


案例3:为虚拟机 server 使用自定Web根目录

调整 Web 站点 http://server0.example.com 的网页目录，要求如下：
1）新建目录 /webroot，作为此站点新的网页目录
2）确保站点 http://server0.example.com  仍然可访问

=[root@server0 ~]# mkdir  /webroot
[root@server0 ~]# cd  /webroot/
[root@server0 webroot]# wget  http://classroom/pub/materials/station.html  -O  index.html
[root@server0 ~]# vim  /etc/httpd/conf.d/00-default.conf
<VirtualHost  *:80>
        ServerName  server0.example.com
        DocumentRoot  /webroot
</VirtualHost>
[root@server0 ~]# systemctl  restart  httpd
[root@desktop0 ~]# elinks  -dump  http://server0.example.com/
[root@server0 ~]# chcon  -R  --reference=/var/www  /webroot/
[root@server0 ~]# ls  -Z  /webroot/index.html 
[root@server0 ~]# vim  /etc/httpd/conf.d/00-default.conf 
<VirtualHost  *:80>
        ServerName  server0.example.com
        DocumentRoot  /webroot
</VirtualHost>
<Directory  "/webroot">
        Require  all  granted
</Directory>
<Directory  "/webroot/private">
        Require  ip  127.0.0.1  ::1  172.25.0.11
</Directory>
[root@server0 ~]# systemctl  restart  httpd 
    [root@server0 ~]# cp  -rf  /var/www/html/private/  /webroot/
[root@desktop0 ~]# elinks  -dump  http://server0.example.com/

案例4：为虚拟机 server 部署动态WSGI站点
为站点 webapp0.example.com 配置提供动态Web内容，要求如下：
1）此虚拟主机侦听在端口8909
2）测试网页从以下地址下载，不要作任何更改http://classroom/pub/materials/webinfo.wsgi 
3）从浏览器访问 http://webapp0.example.com:8909 可接收到动态生成的 Web 页面

= [root@server0 ~]# mkdir  /var/www/webapp0
[root@server0 ~]# cd /var/www/webapp0
[root@server0 webapp0]# wget  http://classroom/pub/materials/webinfo.wsgi
[root@server0 ~]# yum  -y  install  mod_wsgi
    [root@server0 ~]# vim  /etc/httpd/conf.d/02-webapp0.conf
    Listen  8909
    <VirtualHost *:8909>
            DocumentRoot  /var/www/webapp0
            ServerName  webapp0.example.com
            WSGIScriptAlias  /  /var/www/webapp0/webinfo.wsgi
    </VirtualHost>
[root@server0 ~]# semanage  port  -l  |  grep  ^http_port
    [root@server0 ~]# semanage  port  -a  -t  http_port_t  -p  tcp  8909
   [root@server0 ~]# semanage  port  -l  |  grep  ^http_port 
[root@server0 ~]# systemctl  restart  httpd
[root@server0 ~]# netstat  -antpu  |  grep  httpd 
[root@desktop0 ~]# elinks  -dump  http://webapp0.example.com:8909/
